package headFirstJava;
import java.util.ArrayList;
import java.util.Scanner;

public class SimpleDotCom {
    ArrayList<String> locationCells;
    int numOfHits = 0;

    public void setLocationCells(ArrayList<String> locationCells){
        this.locationCells = locationCells;
    }

    public ArrayList<String> makeAim(){
        int rand = (int)(Math.random()*5);
        ArrayList<String> locations = new ArrayList<>();
        locations.add(String.valueOf(rand));
        locations.add(String.valueOf(rand+1));
        locations.add(String.valueOf(rand+2));
        return locations;
    }

    public String checkYourself(String stringGuess){
        int guess = Integer.parseInt(stringGuess);
        String result = "miss";
        for(String cell: locationCells){
            if (String.valueOf(guess).equals(cell)){
                result = "hit";
                numOfHits ++;
                locationCells.remove(cell);
                break;
            }
        }
        if (locationCells.isEmpty()){
            result = "kill";
        }
        System.out.println(result);
        System.out.println("#####################");
        return result;
    }

        int numOfGuess = 0;

    }
}
